☁️ Sincronizar com Google Sheets Grátis

Seus lançamentos são salvos automaticamente numa planilha Google. Funciona em qualquer dispositivo — celular, computador, em qualquer lugar.

  1. Abra o Google Sheets e crie uma planilha nova (ex: "Finanças IA")
  2. No menu: Extensões → Apps Script
  3. Apague o código existente e cole este:
const SHEET_NAME = 'Lançamentos'; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['ID','Descrição','Valor','Tipo','Categoria','Data','Fixo','Dia Fixo']); sheet.getRange(1,1,1,8).setFontWeight('bold').setBackground('#1a472a').setFontColor('#ffffff'); } const data = JSON.parse(e.postData.contents); if (data.action === 'save') { sheet.clearContents(); sheet.appendRow(['ID','Descrição','Valor','Tipo','Categoria','Data','Fixo','Dia Fixo']); sheet.getRange(1,1,1,8).setFontWeight('bold').setBackground('#1a472a').setFontColor('#ffffff'); data.transactions.forEach(t => { sheet.appendRow([t.id,t.description,t.amount,t.type,t.category,t.date,t.isFixed?'Sim':'Não',t.fixedDay||'']); }); } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } } function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return ContentService.createTextOutput(JSON.stringify({transactions:[]})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues().slice(1); const transactions = rows.filter(r=>r[0]).map(r => ({id:r[0],description:r[1],amount:parseFloat(r[2])||0,type:r[3],category:r[4],date:r[5],isFixed:r[6]==='Sim',fixedDay:r[7]||null})); return ContentService.createTextOutput(JSON.stringify({transactions})).setMimeType(ContentService.MimeType.JSON); }
  1. Salve (Ctrl+S) e clique em Implantar → Nova implantação
