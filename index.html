Google Sheets Gratis

Sincroniza todos os dados entre celular e computador.

  1. Acesse sheets.new e crie uma planilha
  2. Menu: Extensoes - Apps Script
  3. Apague tudo e cole o codigo abaixo
  4. Salve - Implantar - Nova implantacao
  5. Tipo: App da Web - Acesso: Qualquer pessoa
  6. Copie a URL e cole abaixo
const SHEET = 'Dados'; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET) || ss.insertSheet(SHEET); const d = JSON.parse(e.postData.contents); if (d.action === 'save') { sh.clearContents(); sh.appendRow(['json']); sh.getRange(2,1).setValue(JSON.stringify(d.data)); } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON); } } function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET); if (!sh || sh.getLastRow() < 2) return ContentService.createTextOutput(JSON.stringify({ok:true,data:null})).setMimeType(ContentService.MimeType.JSON); const json = sh.getRange(2,1).getValue(); return ContentService.createTextOutput(JSON.stringify({ok:true,data:JSON.parse(json)})).setMimeType(ContentService.MimeType.JSON); }
